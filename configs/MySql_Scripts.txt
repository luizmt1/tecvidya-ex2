CREATE DATABASE IF NOT EXISTS 'LOGIN';

SET SQL_MODE="NO_AUTO_VALUE_ON_ZERO";
SET time_zone = "+00:00";

CREATE TABLE IF NOT EXISTS USERS (
  USER_ID BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT 'USER CODE- AUTOINCREMENT',
  USER_LOGIN VARCHAR(64) COLLATE UTF8_UNICODE_CI NOT NULL COMMENT 'NAME',
  USER_PASSWORD_HASH CHAR(118) COLLATE UTF8_UNICODE_CI NOT NULL COMMENT 'HASHED PASSWORD',
  USER_EMAIL VARCHAR(64) COLLATE UTF8_UNICODE_CI NOT NULL COMMENT 'E-MAIL',
  PRIMARY KEY (USER_ID),
  UNIQUE KEY PK_USER_LOGIN (USER_LOGIN)
) ENGINE=MYISAM DEFAULT CHARSET=UTF8 COLLATE=UTF8_UNICODE_CI COMMENT='REGISTERED USERS.' AUTO_INCREMENT=1;
